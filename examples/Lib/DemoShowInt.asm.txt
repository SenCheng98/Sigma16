; Lightweight subroutine.  Doesn't build stack frame, uses R7,R8,R9
; for temporary and R10 for return address

    lea    R7,2[R0]         ; trap write code

    lea    R9,1[R0]         ; write 1 character
    trap   R7,R8,R9         ; write newline
    jump   0[R10]           ; return
